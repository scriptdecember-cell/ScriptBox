-- ── Base64 ───────────────────────────────────────────────────────────────────
local _b64chars='ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/'
local function B64Enc(data)
    return (data:gsub('.',function(x)
        local r,b='',x:byte()
        for i=8,1,-1 do r=r..(b%2^i-b%2^(i-1)>0 and '1' or '0') end
        return r
    end)..'0000'):gsub('%d%d%d%d%d%d',function(x)
        if #x<6 then return '' end
        local c=0
        for i=1,6 do c=c+(x:sub(i,i)=='1' and 2^(6-i) or 0) end
        return _b64chars:sub(c+1,c+1)
    end)..(({'','==','='})[#data%3+1])
end
local function B64Dec(data)
    data=data:gsub('[^'..(_b64chars):gsub('%+','%%+')..'=]','')
    return (data:gsub('.',function(x)
        if x=='=' then return '' end
        local r,f='',(_b64chars:find(x,1,true)-1)
        for i=6,1,-1 do r=r..(f%2^i-f%2^(i-1)>0 and '1' or '0') end
        return r
    end):gsub('%d%d%d%d%d%d%d%d',function(x)
        local c=0
        for i=1,8 do c=c+(x:sub(i,i)=='1' and 2^(8-i) or 0) end
        return string.char(c)
    end))
end

local SHARE_PREFIX="PELMEN1:"

local function GenShareCode()
    local export={}
    for k,v in pairs(Cfg) do
        if type(v)=="table" then
            export[k]={}
            for kk,vv in pairs(v) do
                if type(vv)=="number" or type(vv)=="boolean" then export[k][kk]=vv end
            end
        end
    end
    local ok,json=pcall(HttpService.JSONEncode,HttpService,export)
    if not ok then return nil end
    local ok2,b64=pcall(B64Enc,json)
    if not ok2 then return nil end
    return SHARE_PREFIX..b64
end

local function ImportShareCode(code)
    if type(code)~="string" then return false,"Пустой код" end
    code=code:match("^%s*(.-)%s*$")
    if code:sub(1,#SHARE_PREFIX)~=SHARE_PREFIX then return false,"Неверный код — должен начинаться с PELMEN1:" end
    local b64=code:sub(#SHARE_PREFIX+1)
    local ok,json=pcall(B64Dec,b64)
    if not ok or not json or #json<2 then return false,"Ошибка декодирования" end
    local ok2,parsed=pcall(HttpService.JSONDecode,HttpService,json)
    if not ok2 or type(parsed)~="table" then return false,"Ошибка JSON" end
    for k,v in pairs(parsed) do
        if type(v)=="table" and type(Cfg[k])=="table" then
            for kk,vv in pairs(v) do
                if Cfg[k][kk]~=nil and (type(vv)=="number" or type(vv)=="boolean") then Cfg[k][kk]=vv end
            end
        end
    end
    ScheduleSave()
    return true,"OK"
end

-- ── AI: учимся на имени объекта ───────────────────────────────────────────────
local function aiLearn(name, isThreat)
    if not name or #name<3 then return end
    local n=tostring(name):lower():gsub("[%s_%-]","")

    -- Разбиваем имя на подстроки длиной 4-8 символов (n-gram анализ)
    for len=4,math.min(8,#n) do
        for i=1,#n-len+1 do
            local gram=n:sub(i,i+len-1)
            if isThreat then
                AI.threatMap[gram]=(AI.threatMap[gram] or 0)+1
                -- Если встретили 3+ раз — добавляем в живой словарь
                if AI.threatMap[gram]>=3 and not _suspSet[gram] then
                    _suspSet[gram]=0.6 + AI.threatMap[gram]*0.05
                    AI.learnedSusp[gram]=true
                    AI.generation=AI.generation+1
                end
            else
                -- Встретили в безопасном контексте — снижаем вес
                if AI.threatMap[gram] then
                    AI.threatMap[gram]=AI.threatMap[gram]*0.8
                end
            end
        end
    end
end

-- ── AI: главная функция оценки подозрительности ────────────────────────────────
local function aiScore(name)
    if not name then return 0 end
    local n=tostring(name):lower():gsub("[%s_%-]","")
    local score=0

    -- Прямое совпадение (быстро)
    if _suspSet[n] then return (_suspSet[n]==1 and 10 or _suspSet[n]*10) end

    -- Поиск подстрок (взвешенный)
    for gram,weight in pairs(_suspSet) do
        if n:find(gram,1,true) then
            score=score+weight*(gram:len()/4)
            if score>=5 then return score end -- early exit
        end
    end
    return score
end

local function aiIsSusp(name) return aiScore(name)>=2 end
local function aiIsSuspObj(obj)
    if not obj then return false end
    local ok1,n1=pcall(function() return obj.Name end)
    local ok2,n2=pcall(function() return obj.Parent and obj.Parent.Name end)
    return (ok1 and aiIsSusp(n1)) or (ok2 and aiIsSusp(n2))
end

-- ── AI: адаптация параметров ──────────────────────────────────────────────────
local function aiAdapt()
    local total=AI.stats.total
    if total==0 then return end

    -- Confidence растёт с опытом
    AI.confidence=math.min(100, AI.confidence + total*0.5 + AI.generation*2)

    -- Адаптируем порог TP: если видим много TP-атак — ужесточаем
    local tpRate=AI.stats.tps/math.max(total,1)
    if tpRate>0.3 then
        AI.tpThreshold=math.max(15, (Cfg.Spider.Threshold or 40)*0.7)
    elseif tpRate<0.05 then
        AI.tpThreshold=nil -- откат к дефолту
    end

    -- Адаптируем flood limit: если много флуда — снижаем
    local remoteRate=AI.stats.remotes/math.max(total,1)
    if remoteRate>0.4 then
        AI.floodLimit=math.max(20, AI.floodLimit-5)
    else
        AI.floodLimit=math.min(60, AI.floodLimit+1)
    end

    -- Адаптируем интервал сканирования
    if total>10 then
        AI.scanInterval=math.max(0.5, 2.0/(1+total*0.1))
    end

    AI.generation=AI.generation+1
end

local function spKillConns(obj)
    if not (getconnections and obj) then return 0 end
    local n=0
    -- Только Touched/TouchEnded — самые опасные, остальные редко используются AC
    for _,sig in ipairs({"Touched","TouchEnded","ChildAdded"}) do
        pcall(function()
            for _,c in pairs(getconnections(obj[sig])) do
                if c and c.Connected then
                    pcall(function() c:Disconnect() end) n=n+1
                end
            end
        end)
    end
    return n
end

-- ── СЛОЙ B: Убить скрипты (батчами, не блокирующий) ──────────────────────────
local function spKillScripts(async)
    local srcs={
        game:GetService("ReplicatedStorage"),
        game:GetService("StarterGui"),
        game:GetService("StarterPlayer"),
        Char(),
    }
    local n=0
    local function doKill()
        for _,src in ipairs(srcs) do if src then
            pcall(function()
                for _,scr in pairs(src:GetDescendants()) do
                    if (scr:IsA("LocalScript") or scr:IsA("Script") or scr:IsA("ModuleScript"))
                    and aiIsSusp(scr.Name) then
                        pcall(function() scr.Disabled=true end)
                        aiLearn(scr.Name, true)
                        AI.stats.scripts=AI.stats.scripts+1
                        n=n+1
                    end
                end
            end)
        end end
        return n
    end
    if async then task.spawn(doKill) else doKill() end
    return n
end

-- ── СЛОЙ 1+2: __namecall hook ────────────────────────────────────────────────
local function spHookGameMeta()
    if _SP.ncHooked then return end _SP.ncHooked=true
    pcall(function()
        local mt=getrawmetatable and getrawmetatable(game)
        if not mt then return end
        local oldNC=rawget(mt,"__namecall")
        if setreadonly then setreadonly(mt,false) end

        mt.__namecall=newcclosure and newcclosure(function(self,...)
            local method=(getnamecallmethod and getnamecallmethod()) or ""

            -- Блок Kick
            if method=="Kick" and (self==LP or tostring(self)==tostring(LP)) then
                AI.stats.kicks=AI.stats.kicks+1
                spNotify("🕷️ KICK","Spider заблокировал! #"..AI.stats.kicks,T.Red,4)
                return
            end
            -- Блок Ban/Remove/Destroy на LP
            if (method=="Ban" or method=="Remove" or method=="Destroy") and self==LP then
                spNotify("🕷️ "..method,"Заблокирован",T.Red,4)
                return
            end
            -- Блок Shutdown
            if method=="Shutdown" or method=="Close" then
                spNotify("🕷️ SHUTDOWN","Попытка закрыть игру!",T.Red,5)
                return
            end
            -- Блок FireServer от AC-ремоутов (с AI оценкой)
            if method=="FireServer" or method=="InvokeServer" then
                local ok,isRE=pcall(function() return self:IsA("RemoteEvent") or self:IsA("RemoteFunction") end)
                if ok and isRE then
                    local ok2,name=pcall(function() return self.Name end)
                    if ok2 and aiIsSusp(name) then
                        AI.stats.remotes=AI.stats.remotes+1
                        aiLearn(name, true)
                        return
                    end
                end
            end
            -- Блок серверного телепорта
            if (method=="TeleportToPlaceInstance" or method=="TeleportAsync"
                or method=="Teleport" or method=="TeleportToPrivateServer") and not flyActive then
                AI.stats.tps=AI.stats.tps+1
                spNotify("🕷️ TP SVC","TeleportService заблокирован",T.Red,3)
                return
            end
            return oldNC(self,...)
        end) or function(self,...)
            local method=(getnamecallmethod and getnamecallmethod()) or ""
            if method=="Kick" and self==LP then
                AI.stats.kicks=AI.stats.kicks+1 return
            end
            return oldNC(self,...)
        end

        if setreadonly then setreadonly(mt,true) end
    end)
end

                    if obj:IsA("Script") or obj:IsA("LocalScript") then
                        obj.Disabled=true
                        AI.stats.scripts=AI.stats.scripts+1
                        if score>=5 then
                            spNotify("🕷️ AC Script","Score "..string.format("%.1f",score)..": "..name,T.Yellow,2)
                        end
                    elseif obj:IsA("RemoteEvent") or obj:IsA("RemoteFunction") then
                        spKillConns(obj)
                    end
                else
                    aiLearn(name, false) -- безопасный — учимся
                end

                -- Ragdoll constraint на персонаже
                local ch=Char()
                if (obj:IsA("BallSocketConstraint") or obj:IsA("HingeConstraint"))
                    and ch and obj:IsDescendantOf(ch) then
                    pcall(function() obj.Enabled=false end)
                end
            end)
        end)
    end)
    table.insert(_SP.conns,c)
end

-- ── СЛОЙ 8: Remote flood guard (event-driven, без polling) ───────────────────
local function spRemoteGuard()
    -- Отслеживаем только новые ремоуты через DescendantAdded
    local tracked={}
    local function trackRemote(v)
        if not v:IsA("RemoteEvent") then return end
        if tracked[v] then return end
        -- Уже смотрим AI-оценку
        local ok,name=pcall(function() return v.Name end)
        if ok and aiIsSusp(name) then
            spKillConns(v) aiLearn(name, true) return
        end
        if ok then AI.safeRemotes[name]=true end
        tracked[v]={calls=0,last=tick()}
        local conn conn=v.OnClientEvent:Connect(function()
            if not Cfg.Spider.On then
                pcall(function()conn:Disconnect()end) tracked[v]=nil return
            end
            local t=tracked[v]
            if not t then return end
            local now=tick()
            if now-t.last>1 then t.calls=0 t.last=now end
            t.calls=t.calls+1
            -- AI.floodLimit адаптируется
            if t.calls>AI.floodLimit then
                pcall(function()conn:Disconnect()end)
                tracked[v]=nil
                AI.stats.remotes=AI.stats.remotes+1
                local rname=pcall(function() return v.Name end) and v.Name or "?"
                aiLearn(rname, true)
                spNotify("🕷️ FLOOD","Remote срублен: "..(rname or "?"),T.Yellow,1.5)
            end
        end)
        table.insert(_SP.conns,conn)
        _SP._remoteCache[v]=conn
    end

    -- Трекаем существующие
    task.spawn(function()
        task.wait(0.5)
        for _,v in pairs(game:GetDescendants()) do
            if v:IsA("RemoteEvent") then pcall(function() trackRemote(v) end) end
        end
    end)

    local c=game.DescendantAdded:Connect(function(obj)
        if not Cfg.Spider.On then return end
        pcall(function() trackRemote(obj) end)
    end)
    table.insert(_SP.conns,c)
end

-- ── СЛОЙ 10: Workspace __index spoof ─────────────────────────────────────────
local function spHookWorkspaceMeta()
    pcall(function()
        local mt=getrawmetatable and getrawmetatable(workspace)
        if not mt then return end
        local oldIDX=rawget(mt,"__index")
        if setreadonly then setreadonly(mt,false) end
        mt.__index=newcclosure and newcclosure(function(self,key)
            if Cfg.Spider.On and key=="Gravity" then return _origGravity or 196.2 end
            if type(oldIDX)=="function" then return oldIDX(self,key) end
            return rawget(self,key)
        end) or (type(oldIDX)=="function" and oldIDX or nil)
        if setreadonly then setreadonly(mt,true) end
    end)
end

    end)

-- 🧠 ЕДИНЫЙ HEARTBEAT — нулевой лаг
-- Вместо 10+ отдельных loops — один цикл, работа по расписанию
-- ══════════════════════════════════════════════════════════════════════════════
local _spHB=nil  -- главное соединение heartbeat

local function spStartHeartbeat(h)
    if _spHB then pcall(function() _spHB:Disconnect() end) end

    -- Планировщик задач: каждая задача вызывается по своему интервалу
    local timers={
        -- {interval, last, fn}
        {iv=0.05,  t=0, fn=function()  -- 20fps: критичные проверки
            if not Cfg.Spider.On then return end
            pcall(function()
                -- WalkSpeed
                if Cfg.Speed.On and h and h.Parent then
                    if h.WalkSpeed~=Cfg.Speed.Value then
                        h.WalkSpeed=Cfg.Speed.Value
                    end
                end
                -- JumpPower
                if Cfg.InfJump.On and h and h.Parent then
                    if h.JumpPower~=Cfg.InfJump.Power then
                        h.JumpPower=Cfg.InfJump.Power
                    end
                end
                -- JumpEnabled
                if h and h.Parent and not h.JumpEnabled then h.JumpEnabled=true end
                -- Gravity
                if _origGravity and workspace.Gravity~=_origGravity then
                    workspace.Gravity=_origGravity
                end
            end)
        end},
        {iv=0.1,  t=0, fn=function()  -- 10fps: anti-TP, health
            if not Cfg.Spider.On then return end
            pcall(function()
                local r=Root()
                if not r then _SP._lastSafePos=nil return end
                local cf=r.CFrame
                local threshold=AI.tpThreshold or Cfg.Spider.Threshold or 40
                if _SP._lastSafePos then
                    local dist=(cf.Position-_SP._lastSafePos.Position).Magnitude
                    if dist>threshold and not flyActive then
                        r.CFrame=_SP._lastSafePos
                        AI.stats.tps=AI.stats.tps+1
                        spNotify("🕷️ TP","Anti-TP: "..math.floor(dist).."st",T.Red,1.5)
                        return
                    end
                end
                _SP._lastSafePos=cf
                -- HP guard
                if h and h.Parent and h.Health<h.MaxHealth*0.05 and h.MaxHealth>0 then
                    h.Health=h.MaxHealth
                end
                -- Sit/PlatformStand
                if h and h.Parent then
                    if h.Sit then pcall(function() h.Sit=false end) end
                    if h.PlatformStand and not flyActive then
                        pcall(function() h.PlatformStand=false end)
                    end
                end
            end)
        end},
        {iv=0.3,  t=0, fn=function()  -- 3fps: ragdoll constraints
            if not Cfg.Spider.On then return end
            pcall(function()
                local ch=Char()
                if ch then
                    for _,v in pairs(ch:GetDescendants()) do
                        if (v:IsA("BallSocketConstraint") or v:IsA("HingeConstraint"))
                            and v.Enabled then
                            v.Enabled=false
                        end
                    end
                end
            end)
        end},
        {iv=AI.scanInterval, t=0, fn=function()  -- адаптивный: сканирование + AI обновление
            if not Cfg.Spider.On then return end
            pcall(function()
                -- Скан скриптов
                local now=tick()
                if now-_SP._lastScriptScan>AI.scanInterval then
                    _SP._lastScriptScan=now
                    spKillScripts(true) -- async
                end
                -- AI адаптация
                aiAdapt()
                -- Обновляем интервал этого таймера
                timers[4].iv=AI.scanInterval
            end)
        end},
        {iv=5.0,  t=0, fn=function()  -- каждые 5с: NetworkOwnership
            if not Cfg.Spider.On then return end
            pcall(function()
                local ch=Char() if not ch then return end
                for _,p in pairs(ch:GetDescendants()) do
                    if p:IsA("BasePart") then
                        pcall(function() p:SetNetworkOwner(nil) end)
                    end
                end
            end)
        end},
    }

    _spHB=RunService.Heartbeat:Connect(function(dt)
        if not Cfg.Spider.On then return end
        local now=tick()
        for i=1,#timers do
            local tm=timers[i]
            if now-tm.t>=tm.iv then
                tm.t=now
                tm.fn()
            end
        end
    end)
    table.insert(_SP.conns,_spHB)
end

-- ── AI секундный обучающий тик ────────────────────────────────────────────────
local _aiTick=nil
local function spStartAITick()
    if _aiTick then pcall(function() _aiTick:Disconnect() end) end
    local lastSec=tick()
    _aiTick=RunService.Heartbeat:Connect(function()
        local now=tick()
        if now-lastSec<1 then return end
        lastSec=now
        if not Cfg.Spider.On then return end
        pcall(function()
            -- Каждую секунду: ИИ сканирует потомков на новые паттерны
            local learnt=0
            for _,obj in pairs(game:GetDescendants()) do
                if obj:IsA("LocalScript") or obj:IsA("Script") or obj:IsA("RemoteEvent") then
                    local ok,name=pcall(function() return obj.Name end)
                    if ok and name then
                        local score=aiScore(name)
                        if score>=3 then
                            aiLearn(name, true)
                            learnt=learnt+1
                            if obj:IsA("LocalScript") or obj:IsA("Script") then
                                pcall(function() obj.Disabled=true end)
                            end
                        elseif score<1 then
                            aiLearn(name, false) -- safe whitelist learning
                        end
                    end
                end
            end
            AI.lastLearnTick=tick()
            AI.generation=AI.generation+1
        end)
    end)
    table.insert(_SP.conns,_aiTick)
end

-- ── СТОП / СТАРТ ──────────────────────────────────────────────────────────────
local function StopSpider()
    Cfg.Spider.On=false
    for _,c in pairs(_SP.conns) do pcall(function() c:Disconnect() end) end
    _SP.conns={}
    for obj,mt in pairs(_SP.meta) do pcall(function() setmetatable(obj,mt) end) end
    _SP.meta={}
    _SP.running=false
    _SP._lastSafePos=nil
    _SP._remoteCache={}
    if _aiTick then pcall(function() _aiTick:Disconnect() end) _aiTick=nil end
end

local function StartSpider(h)
    StopSpider()
    if not h then return end
    Cfg.Spider.On=true
    _SP.running=true
    _SP._lastSafePos=nil

    -- Hooks (только один раз, быстро)
    spHookGameMeta()        -- Слой 1+2: __namecall
    spHookHumanoid(h)       -- Слой 3:   __newindex
    spHookHumanoidIndex(h)  -- Слой 4:   __index спуф
    spHookWorkspace()       -- Слой 5:   gravity guard
    spDescMonitor()         -- Слой 6:   DescendantAdded + AI score
    spGuiMonitor()          -- Слой 7:   GUI/Backpack/AntiDestroy
    spRemoteGuard()         -- Слой 8:   flood guard + AI whitelist
    spAnimGuard(h)          -- Слой 9:   animation block
    spHookWorkspaceMeta()   -- Слой 10:  workspace __index spoof
    spHookDataStore()       -- Слой 11:  DataStore ban block
    spHookPlayersMeta()     -- Слой 12:  Players BanAsync

    -- Единый Heartbeat — ВСЕ тяжёлые операции здесь
    spStartHeartbeat(h)     -- Слой 13+: zero-lag scheduler

    -- AI секундный тик
    spStartAITick()         -- 🧠 AI Brain

    -- Первичная зачистка (async, не блокирует)
    task.spawn(function()
        task.wait(0.3)
        local ch=Char()
        if ch then
            for _,p in pairs(ch:GetDescendants()) do
                if p:IsA("BasePart") then spKillConns(p) end
            end
        end
        spKillScripts(false)
    end)

    task.defer(function()
        if type(Notify)=="function" then
            Notify("🕷️ SPIDER v2  TOP-1","AI Brain · Zero-Lag · 12 хуков · ~0 CPU",T.Acc2,5)
        end
    end)
end

-- ── Anti Lag ──────────────────────────────────────────────────────────────────
local _antiLagConns={}
local _antiLagRunning=false

local function StopAntiLag()
    for _,c in pairs(_antiLagConns) do pcall(function()c:Disconnect()end) end
    _antiLagConns={}
    _antiLagRunning=false
end

local function StartAntiLag()
    StopAntiLag()
    _antiLagRunning=true
    task.spawn(function()
        local tracked={}
        while _antiLagRunning and Cfg.AntiLag.On do
            pcall(function()
                for _,v in pairs(game:GetDescendants()) do
                    if v:IsA("RemoteEvent") and not tracked[v] then
                        tracked[v]=true
                        local calls=0 local last=tick()
                        local conn conn=v.OnClientEvent:Connect(function()
                            if not (Cfg.AntiLag.On and _antiLagRunning) then
                                pcall(function()conn:Disconnect()end) tracked[v]=nil return
                            end
                            local now=tick()
                            if now-last>1 then calls=0 last=now end
                            calls=calls+1
                            if calls>30 then
                                pcall(function()conn:Disconnect()end)
                                tracked[v]=nil
                                task.defer(function()
                                    if type(Notify)=="function" then
                                        Notify("🧹 Anti-Lag","Спам Remote срублен: "..v.Name,T.Yellow,1.5)
                                    end
                                end)
                            end
                        end)
                        table.insert(_antiLagConns,conn)
                    end
                end
            end)
            task.wait(2)
        end
    end)
    task.spawn(function()
        while _antiLagRunning and Cfg.AntiLag.On do
            pcall(function()
                for _,v in pairs(game:GetDescendants()) do
                    if v:IsA("LocalScript") or v:IsA("ModuleScript") then
                        local n=v.Name:lower()
                        if n:find("lag") or n:find("flood") or n:find("spam") or n:find("ddos") then
                            pcall(function()v.Disabled=true end)
                        end
                    end
                end
            end)
            task.wait(3)
        end
    end)
    task.spawn(function()
        local beCounts={}
        while _antiLagRunning and Cfg.AntiLag.On do
            pcall(function()
                for _,v in pairs(game:GetDescendants()) do
                    if v:IsA("BindableEvent") and not beCounts[v] then
                        beCounts[v]=0
                        local last=tick()
                        local conn conn=v.Event:Connect(function()
                            if not (Cfg.AntiLag.On and _antiLagRunning) then
                                pcall(function()conn:Disconnect()end) beCounts[v]=nil return
                            end
                            local now=tick()
                            if now-last>1 then beCounts[v]=0 last=now end
                            beCounts[v]=(beCounts[v] or 0)+1
                            if beCounts[v]>50 then
                                pcall(function()conn:Disconnect()end)
                                beCounts[v]=nil
                            end
                        end)
                        table.insert(_antiLagConns,conn)
                    end
                end
            end)
            task.wait(3)
        end
    end)
    task.defer(function()
        if type(Notify)=="function" then Notify("🧹 Anti-Lag","3 слоя защиты включены",T.Yellow,3) end
    end)
end

local function removeESP(player)
    local o=espObjects[player]
    if o then
        pcall(function() o.bg:Destroy() end)
        pcall(function() o.selBox.Adornee=nil end)
        espObjects[player]=nil
    end
end

local function StopESP()
    for p,_ in pairs(espObjects) do removeESP(p) end
    espObjects={}
    for _,c in pairs(espConns) do pcall(function()c:Disconnect()end) end
    espConns={}
end

local function StartESP()
    StopESP()
    for _,p in pairs(Players:GetPlayers()) do
        if p~=LP then
            if p.Character then createESP(p) end
            local c1=p.CharacterAdded:Connect(function()
                task.wait(0.6)
                if Cfg.ESP.On then createESP(p) end
            end)
            table.insert(espConns,c1)
        end
    end
    local c2=Players.PlayerAdded:Connect(function(p)
        if p==LP then return end
        local c1=p.CharacterAdded:Connect(function()
            task.wait(0.6)
            if Cfg.ESP.On then createESP(p) end
        end)
        table.insert(espConns,c1)
    end)
    table.insert(espConns,c2)
    local c3=Players.PlayerRemoving:Connect(function(p) removeESP(p) end)
    table.insert(espConns,c3)
end

local function UpdateESP()
    if not Cfg.ESP.On then return end
    local myRoot=Root()
    local t=tick()
    for player,o in pairs(espObjects) do
        local ok=pcall(function()
            if not (player and player.Parent and player.Character) then
                removeESP(player) return
            end
            local char=player.Character
            local root=char:FindFirstChild("HumanoidRootPart")
            local h=char:FindFirstChildOfClass("Humanoid")
            if not (root and o.bg and o.bg.Parent) then return end

            -- Дистанция
            local dist=0
            if myRoot then dist=math.floor((myRoot.Position-root.Position).Magnitude) end
            o.bg.MaxDistance=Cfg.ESP.MaxDist
            o.bg.Enabled = dist <= Cfg.ESP.MaxDist

            -- Обновляем текст дистанции
            if Cfg.ESP.Dist then
                o.distL.Text="📍 "..dist.." st"
                o.distL.TextColor3=espDistColor(dist)
                o.distL.Visible=true
            else
                o.distL.Visible=false
            end

            -- Имя
            o.nameL.Visible=Cfg.ESP.Names
            o.nameShadow.Visible=Cfg.ESP.Names

            -- Rainbow цвет имени и бокса (уникальный на каждого игрока)
            local hue=(t*0.45+(player.UserId%100)*0.01)%1
            local col=Color3.fromHSV(hue,0.75,1)
            o.nameL.TextColor3=col
            o.cardStroke.Color=col
            o.selBox.Color3=col
            o.selBox.SurfaceColor3=col
            o.avatarBg.BackgroundColor3=col
            o.glow.ImageColor3=col

            -- Пульсирующий glow ближе = ярче
            local glowAlpha=math.clamp(0.3+(1-dist/200)*0.5,0.2,0.75)
            o.glow.ImageTransparency=1-glowAlpha

            -- SelectionBox
            o.selBox.Enabled=Cfg.ESP.Box

            -- HP bar
            if h and Cfg.ESP.Health then
                local pct=math.clamp(h.Health/math.max(h.MaxHealth,1),0,1)
                Tw(o.hpFill,{Size=UDim2.new(pct,0,1,0)},0.12)
                o.hpFill.BackgroundColor3=espHealthColor(h.Health,h.MaxHealth)
                local pctInt=math.floor(pct*100)
                o.hpLabel.Text=pctInt.."%"
                o.hpLabel.TextColor3=espHealthColor(h.Health,h.MaxHealth)
                o.hpGrad.Color=ColorSequence.new{
                    ColorSequenceKeypoint.new(0,espHealthColor(h.Health,h.MaxHealth)),
                    ColorSequenceKeypoint.new(1,Color3.fromHSV(Color3.toHSV(espHealthColor(h.Health,h.MaxHealth)))),
                }
            end
            o.hpFill.Parent.Visible=Cfg.ESP.Health

            -- Нижняя линия пульсирует
            local lineW=0.3+0.7*math.sin(t*2+player.UserId)
            o.bottomLine.BackgroundTransparency=math.clamp(1-lineW*0.9,0.1,0.9)
        end)
    end
end

-- ── Fly system ────────────────────────────────────────────────────────────────
local flyBG=nil local flyBV=nil
local flySpd=0 local flyCtrl={f=0,b=0,l=0,r=0} local flyLastCtrl={f=0,b=0,l=0,r=0}
local mobFlyCtrl={f=false,b=false,l=false,r=false,u=false,d=false}
local function GetFlyMax() return Cfg.Fly.Speed*10 end

local function UpdateFlyCtrl()
    flyCtrl.f=((UIS:IsKeyDown(Enum.KeyCode.W) or UIS:IsKeyDown(Enum.KeyCode.Up))    or mobFlyCtrl.f) and 1  or 0
    flyCtrl.b=((UIS:IsKeyDown(Enum.KeyCode.S) or UIS:IsKeyDown(Enum.KeyCode.Down))  or mobFlyCtrl.b) and -1 or 0
    flyCtrl.l=((UIS:IsKeyDown(Enum.KeyCode.A) or UIS:IsKeyDown(Enum.KeyCode.Left))  or mobFlyCtrl.l) and -1 or 0
    flyCtrl.r=((UIS:IsKeyDown(Enum.KeyCode.D) or UIS:IsKeyDown(Enum.KeyCode.Right)) or mobFlyCtrl.r) and 1  or 0
end

local function StopFly()
    flyActive=false
    local char=Char() if char then
        local h=char:FindFirstChildOfClass("Humanoid") if h then
            for _,s in pairs(Enum.HumanoidStateType:GetEnumItems()) do pcall(function() h:SetStateEnabled(s,true) end) end
            pcall(function() h:ChangeState(Enum.HumanoidStateType.RunningNoPhysics) end)
            h.PlatformStand=false
        end
        local anim=char:FindFirstChild("Animate") if anim then anim.Disabled=false end
    end
    if flyBG and flyBG.Parent then flyBG:Destroy() end flyBG=nil
    if flyBV and flyBV.Parent then flyBV:Destroy() end flyBV=nil
    flySpd=0 flyCtrl={f=0,b=0,l=0,r=0} flyLastCtrl={f=0,b=0,l=0,r=0}
    mobFlyCtrl={f=false,b=false,l=false,r=false,u=false,d=false}
end

local function StartFly()
    if flyActive then StopFly() return end
    local char=Char() if not char then return end
    local h=char:FindFirstChildOfClass("Humanoid") if not h then return end
    local isR6=h.RigType==Enum.HumanoidRigType.R6
    local torso=isR6 and char:FindFirstChild("Torso") or char:FindFirstChild("UpperTorso")
    if not torso then return end
    flyActive=true
    for _,s in pairs(Enum.HumanoidStateType:GetEnumItems()) do pcall(function() h:SetStateEnabled(s,false) end) end
    pcall(function() h:ChangeState(Enum.HumanoidStateType.Swimming) end)
    h.PlatformStand=true
    local anim=char:FindFirstChild("Animate") if anim then anim.Disabled=true end
    for _,tr in pairs(h:GetPlayingAnimationTracks()) do pcall(function() tr:AdjustSpeed(0) end) end
    flyBG=Instance.new("BodyGyro",torso) flyBG.P=9e4 flyBG.MaxTorque=Vector3.new(9e9,9e9,9e9) flyBG.CFrame=torso.CFrame
    flyBV=Instance.new("BodyVelocity",torso) flyBV.Velocity=Vector3.new(0,0.1,0) flyBV.MaxForce=Vector3.new(9e9,9e9,9e9)
    task.spawn(function()
        flySpd=0
        while flyActive and char and h and h.Parent do
            RunService.RenderStepped:Wait() UpdateFlyCtrl()
            local maxS=GetFlyMax()
            local moving=(flyCtrl.l+flyCtrl.r)~=0 or (flyCtrl.f+flyCtrl.b)~=0
            if moving then flySpd=math.min(flySpd+0.6+flySpd/maxS,maxS) else flySpd=math.max(flySpd-1.2,0) end
            local camCF=Camera.CFrame
            if moving then
                flyBV.Velocity=(camCF.LookVector*(flyCtrl.f+flyCtrl.b)+((camCF*CFrame.new(flyCtrl.l+flyCtrl.r,(flyCtrl.f+flyCtrl.b)*0.2,0)).Position-camCF.Position))*flySpd
                flyLastCtrl={f=flyCtrl.f,b=flyCtrl.b,l=flyCtrl.l,r=flyCtrl.r}
            elseif flySpd~=0 then
                flyBV.Velocity=(camCF.LookVector*(flyLastCtrl.f+flyLastCtrl.b)+((camCF*CFrame.new(flyLastCtrl.l+flyLastCtrl.r,(flyLastCtrl.f+flyLastCtrl.b)*0.2,0)).Position-camCF.Position))*flySpd
            else flyBV.Velocity=Vector3.new(0,0,0) end
            if UIS:IsKeyDown(Enum.KeyCode.Space) or mobFlyCtrl.u then
                pcall(function() torso.CFrame=torso.CFrame*CFrame.new(0,1.2,0) end)
            elseif UIS:IsKeyDown(Enum.KeyCode.C) or UIS:IsKeyDown(Enum.KeyCode.LeftControl) or mobFlyCtrl.d then
                pcall(function() torso.CFrame=torso.CFrame*CFrame.new(0,-1.2,0) end)
            end
            pcall(function() flyBG.CFrame=camCF*CFrame.Angles(-math.rad((flyCtrl.f+flyCtrl.b)*45*flySpd/math.max(GetFlyMax(),1)),0,0) end)
        end
        if flyActive then StopFly() end
    end)
end

            end
        end end
        if near then
            local dir=(near.Position-root.Position).Unit

        end
    end
    -- H — быстрый тогл ESP
    if input.KeyCode==Enum.KeyCode.H then
        Cfg.ESP.On=not Cfg.ESP.On ScheduleSave()
        if Cfg.ESP.On then StartESP() else StopESP() end

-- ── On respawn ────────────────────────────────────────────────────────────────
local function ApplyAll(char,isRespawn)
    task.wait(0.7)
    local h=char and char:FindFirstChildOfClass("Humanoid") if not h then return end
    if Cfg.Speed.On   then pcall(function() h.WalkSpeed=Cfg.Speed.Value end) end
    if Cfg.InfJump.On then pcall(function() h.JumpPower=Cfg.InfJump.Power end) end
    if Cfg.AntiRag.On then
        for _,v in pairs(char:GetDescendants()) do
            if v:IsA("BallSocketConstraint") or v:IsA("HingeConstraint") then pcall(function() v.Enabled=false end) end
        end
    end
    if Cfg.Spider.On then StartSpider(h) end
    if Cfg.AntiLag.On then StartAntiLag() end
    if Cfg.ESP.On     then task.delay(0.3,function() if Cfg.ESP.On then StartESP() end end) end
    if flyActive      then StopFly() end
    if isRespawn then
        local feats={}
        if Cfg.Speed.On      then table.insert(feats,"💨"..Cfg.Speed.Value) end
        if Cfg.InfJump.On    then table.insert(feats,"🚀") end
        if Cfg.AntiRag.On    then table.insert(feats,"🛡️") end
        if Cfg.Spider.On     then table.insert(feats,"🕷️") end
        if Cfg.AntiLag.On    then table.insert(feats,"🧹") end
        if Cfg.FastCollect.On then table.insert(feats,"🌿") end
        if Cfg.ESP.On        then table.insert(feats,"👁️") end
        if Cfg.Fullbright.On then table.insert(feats,"💡") end
        if #feats>0 then Notify("🔄 Reload",table.concat(feats," "),T.Acc,3) end
    end
end
