-- ── Base64 ───────────────────────────────────────────────────────────────────
local _b64chars='ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/'
local function B64Enc(data)
    return (data:gsub('.',function(x)
        local r,b='',x:byte()
        for i=8,1,-1 do r=r..(b%2^i-b%2^(i-1)>0 and '1' or '0') end
        return r
    end)..'0000'):gsub('%d%d%d%d%d%d',function(x)
        if #x<6 then return '' end
        local c=0
        for i=1,6 do c=c+(x:sub(i,i)=='1' and 2^(6-i) or 0) end
        return _b64chars:sub(c+1,c+1)
    end)..(({'','==','='})[#data%3+1])
end
local function B64Dec(data)
    data=data:gsub('[^'..(_b64chars):gsub('%+','%%+')..'=]','')
    return (data:gsub('.',function(x)
        if x=='=' then return '' end
        local r,f='',(_b64chars:find(x,1,true)-1)
        for i=6,1,-1 do r=r..(f%2^i-f%2^(i-1)>0 and '1' or '0') end
        return r
    end):gsub('%d%d%d%d%d%d%d%d',function(x)
        local c=0
        for i=1,8 do c=c+(x:sub(i,i)=='1' and 2^(8-i) or 0) end
        return string.char(c)
    end))
end

local SHARE_PREFIX="PELMEN1:"

local function GenShareCode()
    local export={}
    for k,v in pairs(Cfg) do
        if type(v)=="table" then
            export[k]={}
            for kk,vv in pairs(v) do
                if type(vv)=="number" or type(vv)=="boolean" then export[k][kk]=vv end
            end
        end
    end
    local ok,json=pcall(HttpService.JSONEncode,HttpService,export)
    if not ok then return nil end
    local ok2,b64=pcall(B64Enc,json)
    if not ok2 then return nil end
    return SHARE_PREFIX..b64
end

local function ImportShareCode(code)
    if type(code)~="string" then return false,"Пустой код" end
    code=code:match("^%s*(.-)%s*$")
    if code:sub(1,#SHARE_PREFIX)~=SHARE_PREFIX then return false,"Неверный код — должен начинаться с PELMEN1:" end
    local b64=code:sub(#SHARE_PREFIX+1)
    local ok,json=pcall(B64Dec,b64)
    if not ok or not json or #json<2 then return false,"Ошибка декодирования" end
    local ok2,parsed=pcall(HttpService.JSONDecode,HttpService,json)
    if not ok2 or type(parsed)~="table" then return false,"Ошибка JSON" end
    for k,v in pairs(parsed) do
        if type(v)=="table" and type(Cfg[k])=="table" then
            for kk,vv in pairs(v) do
                if Cfg[k][kk]~=nil and (type(vv)=="number" or type(vv)=="boolean") then Cfg[k][kk]=vv end
            end
        end
    end
    ScheduleSave()
    return true,"OK"
end

local function localScore(name)
    if not name then return 0 end
    local n = tostring(name):lower():gsub("[%s_%-]","")
    -- Прямой safe-список
    for k in pairs(_LOCAL_SAFE) do if n:find(k,1,true) then return 0 end end
    -- Прямой threat-список
    local best = 0
    for k, v in pairs(_LOCAL_THREATS) do
        if n:find(k,1,true) then best = math.max(best, v) end
    end
    return best
end

-- ─── GPT API CALL ─────────────────────────────────────────────────────────────
-- Асинхронный вызов GPT, возвращает ответ через callback
local function gptCall(systemPrompt, userMsg, callback, timeoutSec)
    if not HTTP_AVAILABLE then
        callback(nil, "no_http")
        return
    end
    timeoutSec = timeoutSec or 8
    _SP.gptCalls = _SP.gptCalls + 1

    task.spawn(function()
        local ok, result = pcall(function()
            local body = HttpService:JSONEncode({
                model      = OPENAI_MODEL,
                max_tokens = 120,
                temperature= 0.1,  -- почти детерминировано для классификации
                messages   = {
                    {role="system", content=systemPrompt},
                    {role="user",   content=userMsg},
                },
            })
            local response = httpRequest({
                Url     = OPENAI_URL,
                Method  = "POST",
                Headers = {
                    ["Content-Type"]  = "application/json",
                    ["Authorization"] = "Bearer "..OPENAI_KEY,
                },
                Body = body,
            })
            if response.StatusCode ~= 200 then
                return nil, "http_"..tostring(response.StatusCode)
            end
            local parsed = HttpService:JSONDecode(response.Body)
            local text = parsed.choices
                and parsed.choices[1]
                and parsed.choices[1].message
                and parsed.choices[1].message.content
                or nil
            return text
        end)
        if ok and result then
            callback(result, nil)
        else
            _SP.gptErrors = _SP.gptErrors + 1
            spLog("GPT error: "..tostring(result), "error")
            callback(nil, tostring(result))
        end
    end)
end

-- ─── GPT PROMPT: классификация объекта ───────────────────────────────────────
local CLASSIFY_SYSTEM = [[You are an anti-cheat detector for Roblox.
Classify if a script/remote name is from an anti-cheat system trying to detect/ban/kick cheaters.
Reply ONLY with JSON: {"threat":true/false,"confidence":0-10,"reason":"short"}
Be strict: only flag names that are clearly anti-cheat related.
Safe names (GUI, Sound, Shop, Enemy, Weapon, Effect) → threat:false]]

local function gptClassify(name, callback)
    -- Проверяем кэш
    local cached = _SP.gptCache[name]
    if cached and (tick() - cached.time) < _SP.CACHE_TTL then
        callback(cached.result)
        return
    end
    -- Сначала быстрый локальный check
    local localResult = localIsThreat(name)
    -- Если локально очевидно безопасно — не тратим API
    if localScore(name) == 0 then
        _SP.gptCache[name] = {result={threat=false,confidence=0,reason="local_safe",source="local"}, time=tick()}
        callback(_SP.gptCache[name].result)
        return
    end
    -- Запрос к GPT
    gptCall(
        CLASSIFY_SYSTEM,
        'Name: "'..tostring(name)..'"',
        function(text, err)
            if err or not text then
                -- Fallback на локальный
                local res = {threat=localResult, confidence=localResult and 6 or 0, reason="local_fallback", source="local"}
                _SP.gptCache[name] = {result=res, time=tick()}
                callback(res)
                return
            end
            local ok, parsed = pcall(function()
                local clean = text:match("{.-}") or text
                return HttpService:JSONDecode(clean)
            end)
            local res
            if ok and parsed then
                res = {
                    threat    = parsed.threat == true,
                    confidence= tonumber(parsed.confidence) or 5,
                    reason    = parsed.reason or "",
                    source    = "gpt",
                }
            else
                -- GPT вернул не JSON — анализируем текст
                local isThreat = text:lower():find("true") ~= nil
                res = {threat=isThreat, confidence=5, reason="text_parse", source="gpt"}
            end
            _SP.gptCache[name] = {result=res, time=tick()}
            spLog("GPT["..res.source.."]: "..name.." → "..(res.threat and "BLOCK" or "OK").." c="..tostring(res.confidence), res.threat and "block" or "info")
            callback(res)
        end
    )
end

-- ─── GPT БАТЧЕВАЯ ОЧЕРЕДЬ ─────────────────────────────────────────────────────
-- Накапливаем имена → отправляем батч из 10 за раз → экономим API calls
local BATCH_SYSTEM = [[You are an anti-cheat detector for Roblox.
Given a list of script/remote/object names, classify each as anti-cheat (threat) or safe.
Reply ONLY with JSON array: [{"name":"...","threat":true/false,"confidence":0-10}]
Be strict. GUI/Sound/Shop/Tool/Weapon names → safe.]]

local function gptProcessBatch()
    if _SP.gptProcessing or #_SP.gptQueue == 0 then return end
    _SP.gptProcessing = true

    -- Берём до 10 имён
    local batch = {}
    local callbacks = {}
    for i = 1, math.min(10, #_SP.gptQueue) do
        local item = table.remove(_SP.gptQueue, 1)
        if item then
            table.insert(batch, item.name)
            table.insert(callbacks, {name=item.name, cb=item.callback})
        end
    end

    if #batch == 0 then _SP.gptProcessing = false return end

    local namesJson = HttpService:JSONEncode(batch)

    gptCall(
        BATCH_SYSTEM,
        "Names: "..namesJson,
        function(text, err)
            _SP.gptProcessing = false
            if err or not text then
                -- Fallback: отвечаем через локальный классификатор
                for _, item in ipairs(callbacks) do
                    local res = {threat=localIsThreat(item.name), confidence=5, reason="batch_fallback", source="local"}
                    _SP.gptCache[item.name] = {result=res, time=tick()}
                    pcall(function() item.cb(res) end)
                end
                return
            end
            local ok, parsed = pcall(function()
                local clean = text:match("%[.-%]") or text
                return HttpService:JSONDecode(clean)
            end)
            if ok and type(parsed) == "table" then
                -- Строим быстрый lookup по имени
                local lookup = {}
                for _, entry in ipairs(parsed) do
                    if entry.name then lookup[entry.name] = entry end
                end
                for _, item in ipairs(callbacks) do
                    local entry = lookup[item.name]
                    local res
                    if entry then
                        res = {
                            threat    = entry.threat == true,
                            confidence= tonumber(entry.confidence) or 5,
                            reason    = "batch_gpt",
                            source    = "gpt_batch",
                        }
                    else
                        res = {threat=localIsThreat(item.name), confidence=4, reason="not_in_batch", source="local"}
                    end
                    _SP.gptCache[item.name] = {result=res, time=tick()}
                    spLog("BATCH["..res.source.."]: "..item.name.." → "..(res.threat and "BLOCK" or "OK"), res.threat and "block" or "info")
                    pcall(function() item.cb(res) end)
                end
            else
                for _, item in ipairs(callbacks) do
                    local res = {threat=localIsThreat(item.name), confidence=4, reason="batch_parse_fail", source="local"}
                    _SP.gptCache[item.name] = {result=res, time=tick()}
                    pcall(function() item.cb(res) end)
                end
            end
            -- Если ещё есть в очереди — обрабатываем
            if #_SP.gptQueue > 0 then
                task.delay(0.5, gptProcessBatch)
            end
        end,
        12  -- чуть больше timeout для батча
    )
end

local function queueClassify(name, callback)
    -- Быстрый кэш-check
    local cached = _SP.gptCache[name]
    if cached and (tick() - cached.time) < _SP.CACHE_TTL then
        callback(cached.result)
        return
    end
    -- Мгновенный safe-check без API
    if localScore(name) == 0 then
        callback({threat=false, confidence=0, reason="local_safe", source="local"})
        return
    end
    -- Добавляем в очередь
    table.insert(_SP.gptQueue, {name=name, callback=callback})
    -- Запускаем батч с небольшой задержкой (ждём накопления)
    task.delay(0.8, gptProcessBatch)
end

local ADAPT_SYSTEM = [[You are an adaptive anti-cheat configuration system for Roblox.
Analyze the session report and return optimized parameters.
Reply ONLY with JSON:
{
  "tpThreshold": 20-120,
  "floodLimit": 15-80,
  "cacheExpiry": 60-600,
  "aggressiveness": 1-10,
  "notes": "short explanation"
}
aggressiveness: 1=permissive, 10=block everything suspicious]]

local function gptAdapt()
    local now = tick()
    if now - _SP.lastAdaptTime < 30 then return end
    _SP.lastAdaptTime = now

    local sessionAge = math.floor(now - _SP.sessionData.startTime)
    local report = {
        sessionSeconds  = sessionAge,
        kickAttempts    = _SP.sessionData.kickAttempts,
        tpAttempts      = _SP.sessionData.tpAttempts,
        totalBlocked    = _SP.blocked,
        tpBlocked       = _SP.tpBlocked,
        gptCalls        = _SP.gptCalls,

-- ─── СЛОЙ 1+2: __namecall hook ────────────────────────────────────────────────
local function spHookGameMeta()
    if _SP.ncHooked then return end _SP.ncHooked = true
    pcall(function()
        local mt = getrawmetatable and getrawmetatable(game)
        if not mt then return end
        local oldNC = rawget(mt, "__namecall")
        if setreadonly then setreadonly(mt, false) end

        mt.__namecall = newcclosure and newcclosure(function(self, ...)
            local method = (getnamecallmethod and getnamecallmethod()) or ""

            -- KICK — мгновенный блок, синхронно
            if method == "Kick" and (self == LP or tostring(self) == tostring(LP)) then
                _SP.sessionData.kickAttempts = _SP.sessionData.kickAttempts + 1
                spNotify("🕷️ KICK","GPT Spider заблокировал кик!",T.Red,5)
                return
            end
            -- BAN / DESTROY на LP
            if (method=="Ban" or method=="Destroy" or method=="Remove") and self==LP then
                _SP.sessionData.kickAttempts = _SP.sessionData.kickAttempts + 1
                spNotify("🕷️ "..method,"Заблокирован",T.Red,4)
                return
            end
            -- SHUTDOWN
            if method=="Shutdown" or method=="Close" then
                spNotify("🕷️ SHUTDOWN","Закрытие игры заблокировано!",T.Red,5)
                return
            end
            -- FireServer/InvokeServer — GPT async check через очередь
            if method=="FireServer" or method=="InvokeServer" then
                local ok, isRE = pcall(function()
                    return self:IsA("RemoteEvent") or self:IsA("RemoteFunction")
                end)
                if ok and isRE then
                    local ok2, name = pcall(function() return self.Name end)
                    if ok2 and name then
                        -- Сначала быстрый локальный check (без задержки)
                        if localIsThreat(name) then
                            table.insert(_SP.sessionData.remoteNames, 1, name)
                            if #_SP.sessionData.remoteNames > 50 then table.remove(_SP.sessionData.remoteNames) end
                            spLog("Local block FireServer: "..name, "block")
                            return  -- блокируем немедленно
                        end
                        -- Для спорных случаев — ставим в GPT очередь (проверим в фоне)
                        if localScore(name) >= 3 then
                            queueClassify(name, function(res)
                                if res.threat then
                                    table.insert(_SP.sessionData.remoteNames, 1, name)
                                    spLog("GPT queued block: "..name.." ("..tostring(res.confidence)..")", "block")
                                    -- Инвалидируем похожие кэши
                                end
                            end)
                        end
                    end
                end
            end
            -- TELEPORT BLOCK
            if (method=="TeleportToPlaceInstance" or method=="TeleportAsync"
                or method=="Teleport" or method=="TeleportToPrivateServer") and not flyActive then
                _SP.sessionData.tpAttempts = _SP.sessionData.tpAttempts + 1
                spNotify("🕷️ TP SVC","TeleportService заблокирован",T.Red,3)
                return
            end
            -- BanAsync на Players
            if method=="BanAsync" or method=="KickAsync" then
                spNotify("🕷️ BANAYNC","Players:"..method.." заблокирован",T.Red,4)
                return
            end
            return oldNC(self, ...)
        end) or function(self, ...)
            local method = (getnamecallmethod and getnamecallmethod()) or ""
            if method=="Kick" and self==LP then return end
            return oldNC(self, ...)
        end

        if setreadonly then setreadonly(mt, true) end
    end)
end

        local oldNI = rawget(orig, "__newindex") or rawset
        if setreadonly then setreadonly(orig, false) end

            if key=="PlatformStand" and val==true and not flyActive     then return end

            if key=="JumpEnabled"   and val==false                      then return end
            return oldNI and oldNI(self,key,val) or rawset(self,key,val)
        end) or function(self,key,val)
            if key=="WalkSpeed"     then val=Cfg.Speed.On and Cfg.Speed.Value or 16 end
            if key=="JumpPower"     then val=Cfg.InfJump.On and Cfg.InfJump.Power or 50 end
            if key=="PlatformStand" and val==true then return end

        end

    end)

-- ─── СЛОЙ 6: DescendantAdded + GPT очередь ───────────────────────────────────
local function spDescMonitor()
    local c = game.DescendantAdded:Connect(function(obj)
        if not Cfg.Spider.On then return end
        task.defer(function()
            pcall(function()
                local ok, name = pcall(function() return obj.Name end)
                if not ok or not name then return end

                -- Ragdoll cleanup — мгновенно, без GPT
                local ch = Char()
                if (obj:IsA("BallSocketConstraint") or obj:IsA("HingeConstraint"))
                    and ch and obj:IsDescendantOf(ch) then
                    pcall(function() obj.Enabled = false end)
                    return
                end

                -- Для скриптов и ремоутов — GPT classification через батч
                if obj:IsA("Script") or obj:IsA("LocalScript")
                    or obj:IsA("RemoteEvent") or obj:IsA("RemoteFunction") then

                    -- Мгновенный локальный пред-фильтр
                    local sc = localScore(name)
                    if sc == 0 then return end  -- явно безопасно

                    if obj:IsA("Script") or obj:IsA("LocalScript") then
                        table.insert(_SP.sessionData.scriptNames, 1, name)
                        if #_SP.sessionData.scriptNames > 50 then table.remove(_SP.sessionData.scriptNames) end
                    end

                    queueClassify(name, function(res)
                        if not Cfg.Spider.On then return end
                        if res.threat then
                            pcall(function()
                                if obj:IsA("Script") or obj:IsA("LocalScript") then
                                    obj.Disabled = true
                                end
                                if (obj:IsA("RemoteEvent") or obj:IsA("RemoteFunction")) and getconnections then
                                    for _, sig in ipairs({"OnClientEvent","OnServerEvent"}) do
                                        pcall(function()
                                            for _, conn in pairs(getconnections(obj[sig])) do
                                                pcall(function() conn:Disconnect() end)
                                            end
                                        end)
                                    end
                                end
                            end)
                            local src = res.source == "gpt" and "🧠GPT" or res.source == "gpt_batch" and "🧠BATCH" or "📋Local"
                            if res.confidence >= 7 then
                                spNotify("🕷️ "..src,"["..tostring(res.confidence).."] "..name,T.Yellow,2)
                            end
                        end
                    end)
                end
            end)
        end)
    end)
    table.insert(_SP.conns, c)
end

-- ─── СЛОЙ 7: Smart Remote flood guard ────────────────────────────────────────
local function spRemoteGuard()
    local tracked = {}
    local function trackRemote(v)
        if not v:IsA("RemoteEvent") or tracked[v] then return end
        local ok, name = pcall(function() return v.Name end)
        if not ok then return end
        if localScore(name) == 0 then return end  -- явно безопасно

        tracked[v] = {calls=0, last=tick()}
        local conn conn = v.OnClientEvent:Connect(function()
            if not Cfg.Spider.On then
                pcall(function() conn:Disconnect() end) tracked[v]=nil return
            end
            local t = tracked[v]
            if not t then return end
            local now = tick()
            if now - t.last > 1 then t.calls=0 t.last=now end
            t.calls = t.calls + 1
            -- GPT-адаптивный лимит
            if t.calls > _SP.gptFlood then
                pcall(function() conn:Disconnect() end)
                tracked[v] = nil
                table.insert(_SP.sessionData.remoteNames, 1, name)
                spNotify("🕷️ FLOOD","Remote срублен: "..name,T.Yellow,1.5)
                -- Отправляем в GPT очередь для обучения
                queueClassify(name, function(_) end)
            end
        end)
        table.insert(_SP.conns, conn)
    end
    task.spawn(function()
        task.wait(0.5)
        for _, v in pairs(game:GetDescendants()) do
            if v:IsA("RemoteEvent") then pcall(function() trackRemote(v) end) end
        end
    end)
    local c = game.DescendantAdded:Connect(function(obj)
        if Cfg.Spider.On then pcall(function() trackRemote(obj) end) end
    end)
    table.insert(_SP.conns, c)
end

-- ─── ЕДИНЫЙ HEARTBEAT ─────────────────────────────────────────────────────────
local _spHB = nil
local function spStartHeartbeat(h)
    if _spHB then pcall(function() _spHB:Disconnect() end) end
    local sched = {
        -- 20fps: property guards
        {iv=0.05, t=0, fn=function()
            if not Cfg.Spider.On then return end
            pcall(function()
                if Cfg.Speed.On  and h and h.Parent and h.WalkSpeed~=Cfg.Speed.Value  then h.WalkSpeed=Cfg.Speed.Value end
                if Cfg.InfJump.On and h and h.Parent and h.JumpPower~=Cfg.InfJump.Power then h.JumpPower=Cfg.InfJump.Power end
                if h and h.Parent and not h.JumpEnabled then h.JumpEnabled=true end
                if _origGravity and workspace.Gravity~=_origGravity then workspace.Gravity=_origGravity end
            end)
        end},
        -- 10fps: Anti-TP (GPT-адаптивный порог) + health
        {iv=0.1, t=0, fn=function()
            if not Cfg.Spider.On then return end
            pcall(function()
                local r = Root()
                if not r then _SP.lastSafePos=nil return end
                local cf = r.CFrame
                -- Используем GPT-адаптированный порог если есть
                local threshold = _SP.gptThreshold or Cfg.Spider.Threshold or 40
                if _SP.lastSafePos then
                    local dist = (cf.Position - _SP.lastSafePos.Position).Magnitude
                    if dist > threshold and not flyActive then
                        r.CFrame = _SP.lastSafePos
                        _SP.tpBlocked = _SP.tpBlocked + 1
                        -- Записываем событие для GPT адаптации
                        table.insert(_SP.tpEvents, 1, {dist=math.floor(dist), time=tick()})
                        if #_SP.tpEvents > 30 then table.remove(_SP.tpEvents) end
                        _SP.sessionData.tpAttempts = _SP.sessionData.tpAttempts + 1
                        spLog("Anti-TP: "..math.floor(dist).."st (порог "..threshold.."st)", "block")
                        spNotify("🕷️ ANTI-TP","Откат заблокирован · "..math.floor(dist).."st",T.Red,2)
                        return
                    end
                end
                _SP.lastSafePos = cf
                -- HP
                if h and h.Parent and h.Health < h.MaxHealth*0.05 and h.MaxHealth > 0 then
                    h.Health = h.MaxHealth
                end
                if h and h.Parent then
                    if h.Sit then pcall(function() h.Sit=false end) end
                    if h.PlatformStand and not flyActive then pcall(function() h.PlatformStand=false end) end
                end
            end)
        end},
        -- 4fps: ragdoll cleanup
        {iv=0.25, t=0, fn=function()
            if not Cfg.Spider.On then return end
            pcall(function()
                local ch = Char() if not ch then return end
                for _, v in pairs(ch:GetDescendants()) do
                    if (v:IsA("BallSocketConstraint") or v:IsA("HingeConstraint")) and v.Enabled then
                        v.Enabled = false
                    end
                end
            end)
        end},
        -- каждые 30с: GPT адаптация (только когда API доступен)
        {iv=30.0, t=0, fn=function()
            if not Cfg.Spider.On or not HTTP_AVAILABLE then return end
            pcall(gptAdapt)
        end},
        -- каждые 10с: обработка GPT очереди если накопилась
        {iv=10.0, t=0, fn=function()
            if not Cfg.Spider.On or not HTTP_AVAILABLE then return end
            pcall(gptProcessBatch)
        end},
        -- каждые 8с: NetworkOwnership
        {iv=8.0, t=0, fn=function()
            if not Cfg.Spider.On then return end
            pcall(function()
                local ch = Char() if not ch then return end
                for _, p in pairs(ch:GetDescendants()) do
                    if p:IsA("BasePart") then pcall(function() p:SetNetworkOwner(nil) end) end
                end
            end)
        end},
    }
    _spHB = RunService.Heartbeat:Connect(function()
        if not Cfg.Spider.On then return end
        local now = tick()
        for i = 1, #sched do
            local t = sched[i]
            if now - t.t >= t.iv then
                t.t = now
                t.fn()
            end
        end
    end)
    table.insert(_SP.conns, _spHB)
end

-- ─── STOP / START ─────────────────────────────────────────────────────────────
local function StopSpider()
    Cfg.Spider.On = false
    for _, c in pairs(_SP.conns) do pcall(function() c:Disconnect() end) end
    _SP.conns = {}
    for obj, mt in pairs(_SP.meta) do pcall(function() setmetatable(obj,mt) end) end
    _SP.meta = {}
    _SP.running = false
    _SP.lastSafePos = nil
    _SP.gptQueue = {}
    _SP.gptProcessing = false
    if _spHB then pcall(function() _spHB:Disconnect() end) _spHB=nil end
end

local function StartSpider(h)
    StopSpider()
    if not h then return end
    Cfg.Spider.On = true
    _SP.running = true
    _SP.lastSafePos = nil
    _SP.sessionData.startTime = tick()

    spHookGameMeta()      -- Слой 1+2
    spHookHumanoid(h)     -- Слой 3
    spHookHumanoidIndex(h)-- Слой 4
    spHookWorkspace()     -- Слой 5
    spDescMonitor()       -- Слой 6
    spRemoteGuard()       -- Слой 7
    spGuiProtect()        -- Слой 8
    spStartHeartbeat(h)   -- Планировщик + Anti-TP

    -- Первичный scan (async)
    task.spawn(function()
        task.wait(0.5)
        for _, obj in pairs(game:GetDescendants()) do
            if (obj:IsA("LocalScript") or obj:IsA("Script")) and localScore((pcall(function() return obj.Name end)) and obj.Name or "") >= 6 then
                pcall(function() obj.Disabled=true end)
            end
        end
    end)

    -- Первый GPT адаптационный вызов через 5с
    if HTTP_AVAILABLE then
        task.delay(5, function()
            if Cfg.Spider.On then gptAdapt() end
        end)
    end

    local apiMsg = HTTP_AVAILABLE and "GPT-4o-mini ONLINE" or "⚠️ API недоступен · только Local"
    task.defer(function()
        if type(Notify)=="function" then
            Notify("🕷️ SPIDER v1","Anti-AC · Anti-TP · "..apiMsg,T.Acc2,5)
        end
    end)
    spLog("Spider v1 started · "..apiMsg, "info")
end

                                        Notify("🧹 Anti-Lag","Спам Remote срублен: "..v.Name,T.Yellow,1.5)

                    end
                end
            end)

        end
    end)
    task.spawn(function()
        while _antiLagRunning and Cfg.AntiLag.On do
            pcall(function()
                for _,v in pairs(game:GetDescendants()) do
                    if v:IsA("LocalScript") or v:IsA("ModuleScript") then

                            if not (Cfg.AntiLag.On and _antiLagRunning) then
                                pcall(function()conn:Disconnect()end) beCounts[v]=nil return

local _origLight={}
local function StartFullbright()
    _origLight.Brightness       = Lighting.Brightness
    _origLight.Ambient          = Lighting.Ambient
    _origLight.OutdoorAmbient   = Lighting.OutdoorAmbient
    _origLight.FogEnd           = Lighting.FogEnd
    _origLight.FogStart         = Lighting.FogStart
    _origLight.ClockTime        = Lighting.ClockTime
    pcall(function()
        Lighting.Brightness     = 2
        Lighting.Ambient        = Color3.new(1,1,1)
        Lighting.OutdoorAmbient = Color3.new(1,1,1)
        Lighting.FogEnd         = 100000
        Lighting.FogStart       = 100000
        Lighting.ClockTime      = 14
        -- Отключаем Atmosphere и ColorCorrection
        for _,v in pairs(Lighting:GetChildren()) do
            if v:IsA("Atmosphere") or v:IsA("ColorCorrectionEffect") or v:IsA("BloomEffect") then
                v.Enabled=false
            end
        end
    end)
end

local function StopFullbright()
    pcall(function()
        Lighting.Brightness     = _origLight.Brightness     or 1
        Lighting.Ambient        = _origLight.Ambient        or Color3.new(0,0,0)
        Lighting.OutdoorAmbient = _origLight.OutdoorAmbient or Color3.new(0.5,0.5,0.5)
        Lighting.FogEnd         = _origLight.FogEnd         or 100000
        Lighting.FogStart       = _origLight.FogStart       or 0
        Lighting.ClockTime      = _origLight.ClockTime      or 14
        for _,v in pairs(Lighting:GetChildren()) do
            if v:IsA("Atmosphere") or v:IsA("ColorCorrectionEffect") or v:IsA("BloomEffect") then
                v.Enabled=true
            end
        end
    end)
end

local function espHealthColor(h,maxH)
    local pct=math.clamp(h/math.max(maxH,1),0,1)
    if pct>0.6 then return Color3.fromRGB(80,220,120)
    elseif pct>0.3 then return Color3.fromRGB(240,200,60)
    else return Color3.fromRGB(220,70,70) end
end

local function espDistColor(dist)
    local pct=math.clamp(dist/200,0,1)
    -- близко = ярко-красный, далеко = фиолетовый
    local r=math.floor(220*(1-pct)+110*pct)
    local g=math.floor(70*(1-pct)+80*pct)
    local b=math.floor(70*(1-pct)+255*pct)
    return Color3.fromRGB(r,g,b)
end

local function removeESP(player)
    local o=espObjects[player]
    if o then
        pcall(function() o.bg:Destroy() end)
        pcall(function() o.selBox.Adornee=nil end)
        espObjects[player]=nil
    end
end

local function StopESP()
    for p,_ in pairs(espObjects) do removeESP(p) end
    espObjects={}
    for _,c in pairs(espConns) do pcall(function()c:Disconnect()end) end
    espConns={}
end

local function StartESP()
    StopESP()
    for _,p in pairs(Players:GetPlayers()) do
        if p~=LP then
            if p.Character then createESP(p) end
            local c1=p.CharacterAdded:Connect(function()
                task.wait(0.6)
                if Cfg.ESP.On then createESP(p) end
            end)
            table.insert(espConns,c1)
        end
    end
    local c2=Players.PlayerAdded:Connect(function(p)
        if p==LP then return end
        local c1=p.CharacterAdded:Connect(function()
            task.wait(0.6)
            if Cfg.ESP.On then createESP(p) end
        end)
        table.insert(espConns,c1)
    end)
    table.insert(espConns,c2)
    local c3=Players.PlayerRemoving:Connect(function(p) removeESP(p) end)
    table.insert(espConns,c3)
end

local function UpdateESP()
    if not Cfg.ESP.On then return end
    local myRoot=Root()
    local t=tick()
    for player,o in pairs(espObjects) do
        local ok=pcall(function()
            if not (player and player.Parent and player.Character) then
                removeESP(player) return
            end
            local char=player.Character
            local root=char:FindFirstChild("HumanoidRootPart")
            local h=char:FindFirstChildOfClass("Humanoid")
            if not (root and o.bg and o.bg.Parent) then return end

            -- Дистанция
            local dist=0
            if myRoot then dist=math.floor((myRoot.Position-root.Position).Magnitude) end
            o.bg.MaxDistance=Cfg.ESP.MaxDist
            o.bg.Enabled = dist <= Cfg.ESP.MaxDist

            -- Обновляем текст дистанции
            if Cfg.ESP.Dist then
                o.distL.Text="📍 "..dist.." st"
                o.distL.TextColor3=espDistColor(dist)
                o.distL.Visible=true
            else
                o.distL.Visible=false
            end

            -- Имя
            o.nameL.Visible=Cfg.ESP.Names
            o.nameShadow.Visible=Cfg.ESP.Names

            -- Rainbow цвет имени и бокса (уникальный на каждого игрока)
            local hue=(t*0.45+(player.UserId%100)*0.01)%1
            local col=Color3.fromHSV(hue,0.75,1)
            o.nameL.TextColor3=col
            o.cardStroke.Color=col
            o.selBox.Color3=col
            o.selBox.SurfaceColor3=col
            o.avatarBg.BackgroundColor3=col
            o.glow.ImageColor3=col

            -- Пульсирующий glow ближе = ярче
            local glowAlpha=math.clamp(0.3+(1-dist/200)*0.5,0.2,0.75)
            o.glow.ImageTransparency=1-glowAlpha

            -- SelectionBox
            o.selBox.Enabled=Cfg.ESP.Box

            -- HP bar
            if h and Cfg.ESP.Health then
                local pct=math.clamp(h.Health/math.max(h.MaxHealth,1),0,1)
                Tw(o.hpFill,{Size=UDim2.new(pct,0,1,0)},0.12)
                o.hpFill.BackgroundColor3=espHealthColor(h.Health,h.MaxHealth)
                local pctInt=math.floor(pct*100)
                o.hpLabel.Text=pctInt.."%"
                o.hpLabel.TextColor3=espHealthColor(h.Health,h.MaxHealth)
                o.hpGrad.Color=ColorSequence.new{
                    ColorSequenceKeypoint.new(0,espHealthColor(h.Health,h.MaxHealth)),
                    ColorSequenceKeypoint.new(1,Color3.fromHSV(Color3.toHSV(espHealthColor(h.Health,h.MaxHealth)))),
                }
            end
            o.hpFill.Parent.Visible=Cfg.ESP.Health

            -- Нижняя линия пульсирует
            local lineW=0.3+0.7*math.sin(t*2+player.UserId)
            o.bottomLine.BackgroundTransparency=math.clamp(1-lineW*0.9,0.1,0.9)
        end)
    end
end

-- ── Fly system ────────────────────────────────────────────────────────────────
local flyBG=nil local flyBV=nil
local flySpd=0 local flyCtrl={f=0,b=0,l=0,r=0} local flyLastCtrl={f=0,b=0,l=0,r=0}
local mobFlyCtrl={f=false,b=false,l=false,r=false,u=false,d=false}
local function GetFlyMax() return Cfg.Fly.Speed*10 end

local function UpdateFlyCtrl()
    flyCtrl.f=((UIS:IsKeyDown(Enum.KeyCode.W) or UIS:IsKeyDown(Enum.KeyCode.Up))    or mobFlyCtrl.f) and 1  or 0
    flyCtrl.b=((UIS:IsKeyDown(Enum.KeyCode.S) or UIS:IsKeyDown(Enum.KeyCode.Down))  or mobFlyCtrl.b) and -1 or 0
    flyCtrl.l=((UIS:IsKeyDown(Enum.KeyCode.A) or UIS:IsKeyDown(Enum.KeyCode.Left))  or mobFlyCtrl.l) and -1 or 0
    flyCtrl.r=((UIS:IsKeyDown(Enum.KeyCode.D) or UIS:IsKeyDown(Enum.KeyCode.Right)) or mobFlyCtrl.r) and 1  or 0
end

local function StopFly()
    flyActive=false
    local char=Char() if char then
        local h=char:FindFirstChildOfClass("Humanoid") if h then
            for _,s in pairs(Enum.HumanoidStateType:GetEnumItems()) do pcall(function() h:SetStateEnabled(s,true) end) end
            pcall(function() h:ChangeState(Enum.HumanoidStateType.RunningNoPhysics) end)
            h.PlatformStand=false
        end
        local anim=char:FindFirstChild("Animate") if anim then anim.Disabled=false end
    end
    if flyBG and flyBG.Parent then flyBG:Destroy() end flyBG=nil
    if flyBV and flyBV.Parent then flyBV:Destroy() end flyBV=nil
    flySpd=0 flyCtrl={f=0,b=0,l=0,r=0} flyLastCtrl={f=0,b=0,l=0,r=0}
    mobFlyCtrl={f=false,b=false,l=false,r=false,u=false,d=false}
end

local function StartFly()
    if flyActive then StopFly() return end
    local char=Char() if not char then return end
    local h=char:FindFirstChildOfClass("Humanoid") if not h then return end
    local isR6=h.RigType==Enum.HumanoidRigType.R6
    local torso=isR6 and char:FindFirstChild("Torso") or char:FindFirstChild("UpperTorso")
    if not torso then return end
    flyActive=true
    for _,s in pairs(Enum.HumanoidStateType:GetEnumItems()) do pcall(function() h:SetStateEnabled(s,false) end) end
    pcall(function() h:ChangeState(Enum.HumanoidStateType.Swimming) end)
    h.PlatformStand=true
    local anim=char:FindFirstChild("Animate") if anim then anim.Disabled=true end
    for _,tr in pairs(h:GetPlayingAnimationTracks()) do pcall(function() tr:AdjustSpeed(0) end) end
    flyBG=Instance.new("BodyGyro",torso) flyBG.P=9e4 flyBG.MaxTorque=Vector3.new(9e9,9e9,9e9) flyBG.CFrame=torso.CFrame
    flyBV=Instance.new("BodyVelocity",torso) flyBV.Velocity=Vector3.new(0,0.1,0) flyBV.MaxForce=Vector3.new(9e9,9e9,9e9)
    task.spawn(function()
        flySpd=0
        while flyActive and char and h and h.Parent do
            RunService.RenderStepped:Wait() UpdateFlyCtrl()
            local maxS=GetFlyMax()
            local moving=(flyCtrl.l+flyCtrl.r)~=0 or (flyCtrl.f+flyCtrl.b)~=0
            if moving then flySpd=math.min(flySpd+0.6+flySpd/maxS,maxS) else flySpd=math.max(flySpd-1.2,0) end
            local camCF=Camera.CFrame
            if moving then
                flyBV.Velocity=(camCF.LookVector*(flyCtrl.f+flyCtrl.b)+((camCF*CFrame.new(flyCtrl.l+flyCtrl.r,(flyCtrl.f+flyCtrl.b)*0.2,0)).Position-camCF.Position))*flySpd
                flyLastCtrl={f=flyCtrl.f,b=flyCtrl.b,l=flyCtrl.l,r=flyCtrl.r}
            elseif flySpd~=0 then
                flyBV.Velocity=(camCF.LookVector*(flyLastCtrl.f+flyLastCtrl.b)+((camCF*CFrame.new(flyLastCtrl.l+flyLastCtrl.r,(flyLastCtrl.f+flyLastCtrl.b)*0.2,0)).Position-camCF.Position))*flySpd
            else flyBV.Velocity=Vector3.new(0,0,0) end
            if UIS:IsKeyDown(Enum.KeyCode.Space) or mobFlyCtrl.u then
                pcall(function() torso.CFrame=torso.CFrame*CFrame.new(0,1.2,0) end)
            elseif UIS:IsKeyDown(Enum.KeyCode.C) or UIS:IsKeyDown(Enum.KeyCode.LeftControl) or mobFlyCtrl.d then
                pcall(function() torso.CFrame=torso.CFrame*CFrame.new(0,-1.2,0) end)
            end
            pcall(function() flyBG.CFrame=camCF*CFrame.Angles(-math.rad((flyCtrl.f+flyCtrl.b)*45*flySpd/math.max(GetFlyMax(),1)),0,0) end)
        end
        if flyActive then StopFly() end
    end)
end

-- ── On respawn ────────────────────────────────────────────────────────────────
local function ApplyAll(char,isRespawn)
    task.wait(0.7)
    local h=char and char:FindFirstChildOfClass("Humanoid") if not h then return end
    if Cfg.Speed.On   then pcall(function() h.WalkSpeed=Cfg.Speed.Value end) end
    if Cfg.InfJump.On then pcall(function() h.JumpPower=Cfg.InfJump.Power end) end
    if Cfg.AntiRag.On then
        for _,v in pairs(char:GetDescendants()) do
            if v:IsA("BallSocketConstraint") or v:IsA("HingeConstraint") then pcall(function() v.Enabled=false end) end
        end
    end
    if Cfg.Spider.On then StartSpider(h) end
    if Cfg.AntiLag.On then StartAntiLag() end
    if Cfg.ESP.On     then task.delay(0.3,function() if Cfg.ESP.On then StartESP() end end) end
    if flyActive      then StopFly() end
    if isRespawn then
        local feats={}
        if Cfg.Speed.On      then table.insert(feats,"💨"..Cfg.Speed.Value) end
        if Cfg.InfJump.On    then table.insert(feats,"🚀") end
        if Cfg.AntiRag.On    then table.insert(feats,"🛡️") end

-- ── ShiftLock ─────────────────────────────────────────────────────────────────
RunService.RenderStepped:Connect(function()
    local char=Char() if not char then return end
    local root=char:FindFirstChild("HumanoidRootPart")
    local h=char:FindFirstChildOfClass("Humanoid")
    if not (root and h) then return end
    if Cfg.ShiftLock.On then
        UIS.MouseBehavior=Enum.MouseBehavior.LockCenter
        local lv=Camera.CFrame.LookVector
        local xz=Vector3.new(lv.X,0,lv.Z)
        if xz.Magnitude>0.01 then
            pcall(function() root.CFrame=root.CFrame:Lerp(CFrame.new(root.CFrame.Position,root.CFrame.Position+xz),0.6) end)
        end
        h.CameraOffset=h.CameraOffset:Lerp(Vector3.new(1.75,0,0),0.2)
    else
        if UIS.MouseBehavior==Enum.MouseBehavior.LockCenter then UIS.MouseBehavior=Enum.MouseBehavior.Default end
        h.CameraOffset=h.CameraOffset:Lerp(Vector3.new(0,0,0),0.2)
    end
end)

    local t=tick()
    for _,item in ipairs(_rainbow) do
        if item.stroke then
            pcall(function()
                if item.stroke and item.stroke.Parent then
                    item.stroke.Color=rainbowColor(t,item.shift or 0,item.sat or 0.72,0.95)
                end
            end)
        elseif item.grad then
            pcall(function()
                if item.grad and item.grad.Parent then
                    local s=item.shift or 0
                    local sp2=item.speed or 1
                    local function bw(off) local vv=0.5+0.5*math.sin(t*sp2*1.4+(s+off)*6.2832) return Color3.new(0.15+(0.8*vv),0.15+(0.8*vv),0.15+(0.8*vv)) end
                    item.grad.Color=ColorSequence.new{
                        ColorSequenceKeypoint.new(0,   bw(0)),
                        ColorSequenceKeypoint.new(0.5, bw(0.18)),
                        ColorSequenceKeypoint.new(1,   bw(0.36)),
                    }
                end
            end)
        end
    end
end)
